rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Users collection - allow reading and querying for username checks, writing for own data
    match /users/{userId} {
      allow read, list: if true; // Allow reading and querying for username checks during login
      allow write: if request.auth != null && request.auth.uid == userId;
      allow create: if request.auth != null;
    }
    
    // Clients can read/write their own data
    match /clients/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Employees can read/write their own data and read all employees for messaging
    match /employees/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
      allow read: if request.auth != null; // Allow employees to read other employees for messaging
    }
    
    // Admins can read/write their own data
    match /admins/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Bookings - temporary open access for development
    match /bookings/{bookingId} {
      allow read: if true; // Temporary for development
      allow write: if request.auth != null;
    }
    
    // Jobs - role-based access control
    match /jobs/{jobId} {
      // Role-based read access
      allow read: if request.auth != null && (
        request.auth.uid == resource.data.clientId || // Client owns job
        request.auth.uid == resource.data.employeeId || // Employee assigned to job
        (exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
         get(/databases/$(database)/documents/users/$(request.auth.uid)).data.userType == 'admin') // Admin
      );
      
      // Allow unauthenticated read for guest jobs
      allow read: if resource.data.clientId == null;
      
      // List permission for querying
      allow list: if request.auth != null;
      
      allow create: if true; // Allow guests to create jobs
      allow write: if request.auth != null;
    }
    
    // Messages - authenticated users can read/write/create/list (includes image messages)
    match /messages/{messageId} {
      allow read, write, create, list: if request.auth != null;
    }
    
    // Conversations - authenticated users can read/write/create/list
    match /conversations/{conversationId} {
      allow read, write, create, list: if request.auth != null;
    }
    
    // Guest conversations - allow unauthenticated access for guest support
    match /guest-conversations/{conversationId} {
      allow read, write, create, list: if true;
    }
    
    // Guest messages - allow unauthenticated access for guest support
    match /guest-messages/{messageId} {
      allow read, write, create, list: if true;
    }
    
    // Work requests
    match /workRequests/{requestId} {
      allow read, write: if request.auth != null;
    }
    
    // Guest bookings - allow creation and reading
    match /guest-bookings/{bookingId} {
      allow create, read: if true;
      allow update, delete: if request.auth != null;
    }
    
    // Audit entries - authenticated users can read/write
    match /audit/{auditId} {
      allow read, write: if request.auth != null;
    }
    
    // News - authenticated users can read/write
    match /news/{newsId} {
      allow read, write: if request.auth != null;
    }
  }
}

// Firebase Storage Rules
service firebase.storage {
  match /b/{bucket}/o {
    
    // Job attachments (10MB limit)
    match /job-attachments/{jobId}/{fileName} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.resource.size < 10 * 1024 * 1024;
    }

    // Job photos (5MB limit)
    match /job-photos/{jobId}/{fileName} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.resource.size < 5 * 1024 * 1024;
    }

    // Profile images (2MB limit, must match userId)
    match /profile-images/{userId}/{fileName} {
      allow read: if request.auth != null;
      allow write: if request.auth != null 
        && request.auth.uid == userId
        && request.resource.size < 2 * 1024 * 1024;
    }
    // Simpler fallback
    match /profile-images/{fileName} {
      allow read, write: if request.auth != null;
    }

    // Chat images (5MB limit)
    match /chat-images/{chatId}/{fileName} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.resource.size < 5 * 1024 * 1024;
    }
    // Simpler fallback
    match /chat-images/{userId}/{fileName} {
      allow read, write: if request.auth != null;
    }

    // Equipment photos (5MB limit)
    match /equipment-photos/{equipmentId}/{fileName} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.resource.size < 5 * 1024 * 1024;
    }

    // Message images (general, no size restriction)
    match /message-images/{allPaths=**} {
      allow read, write: if request.auth != null;
    }

    // Fallback: allow authenticated uploads under 10MB
    match /{allPaths=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.resource.size < 10 * 1024 * 1024;
    }
  }
}

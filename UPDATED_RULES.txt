rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Users can read/write their own user document
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Users can read/write their own settings
    match /userSettings/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Authenticated users can read/write jobs
    match /jobs/{jobId} {
      allow read, write: if request.auth != null;
    }
    
    // Allow admins to read all users (for admin dashboard)
    match /users/{userId} {
      allow read: if request.auth != null && 
        exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.userType == 'admin';
    }
    
    // Guest bookings - anyone can create, admin can manage
    match /guest-bookings/{bookingId} {
      allow create: if true;
      allow read, update, delete: if request.auth != null && 
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.userType == 'admin';
    }
    
    // Allow authenticated users to create/read their own client profile
    match /clients/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Allow authenticated users to read their own employee profile
    match /employees/{userId} {
      allow read: if request.auth != null && request.auth.uid == userId;
    }
    
    // Allow authenticated users to read their own admin profile
    match /admins/{userId} {
      allow read: if request.auth != null && request.auth.uid == userId;
    }
    
    // Allow authenticated users to create/read bookings
    match /bookings/{bookingId} {
      allow read, write: if request.auth != null;
    }
    
    // Work requests collection
    match /workRequests/{requestId} {
      allow read, write: if request.auth != null;
    }
    
    // Messages collection - authenticated users can read/write
    match /messages/{messageId} {
      allow read, write: if request.auth != null;
    }
    
    // News collection - authenticated users can read/write
    match /news/{newsId} {
      allow read, write: if request.auth != null;
    }
  }
}
